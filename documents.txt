1) git clone https://gitlab.com/megtag/mongo-seeds.git
2) npm install
3) node app --options

--options 
	--host  ip {default:127.0.0.1}
	--port  port {default:27017}
	--db dbName {default:allsec}
	--collection collectionName {default:intentResponses}
	--appId appId {default: new objectId}
	--queryId cloneAppId *required
example :
	node app --host 192.168.0.101 --port 20107 --db allsec --collection intentResponses --appId 593e43d98e90fc2a459d1f85 --queryId 593e43d98e90fc2a459d1f84

4)If any problem occures use following command
	node app --restore

	note: its only restore the last backup collection data
